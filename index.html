📦 Installation

Use CDN or install via npm.

import juiceToast from "https://cdn.jsdelivr.net/npm/juice-toast/dist/juice-toast.esm.js";
<script src="https://cdn.jsdelivr.net/npm/juice-toast/dist/juice-toast.umd.js"></script>
npm install juice-toast

🚀 Quick Start

juiceToast.setup({
  duration: 5000,
  maxVisible: 3,
  glassUI: true
});

juiceToast.success({
  title: "Success",
  message: "Saved successfully",
  size: "lg",
  width: "360px"
});

Call `juiceToast.success`, `.error`, `.custom` to show toasts.

📝 Payload Example

🎬 Animation

juiceToast.success({
  animation: "slide-in",
  enterAnimation: "bounce"
});

🎨 Themes

juiceToast.defineTheme("ocean",{
  bg: "#0ea5e9",
  color: "#fff"
});

🔌 Plugin System

juiceToast.use(ctx => {
  console.log(ctx.type);
});
