📦 Installation

Use CDN (ESM) or install via npm.

import juiceToast from "https://cdn.jsdelivr.net/npm/juice-toast@1.3.2/dist/juice-toast.esm.js";
npm install juice-toast --save

🚀 Quick Start

juiceToast.setup({
  duration: 5000,
  maxVisible: 3,
  glassUI: true
});

juiceToast.success({
  title: "Success",
  message: "Saved successfully",
  size: "lg",
  width: "360px"
});

Call `juiceToast.success`, `.error`, `.custom` to show toasts.

📝 Payload Example

juiceToast.success({
  title: "Success",
  message: "Hello world",
  progress: true,
  closable: true,
  size: "lg",
  width: "380px"
});

🎬 Animation

juiceToast.success({
  animation: "slide-in",
  enterAnimation: "bounce"
});

🎨 Themes

juiceToast.defineTheme("ocean",{
  bg: "#0ea5e9",
  color: "#fff"
});

🔌 Plugin System

juiceToast.use(ctx => {
  console.log(ctx.type);
});

🧹 Utilities

juiceToast.clear();
juiceToast.destroy();
